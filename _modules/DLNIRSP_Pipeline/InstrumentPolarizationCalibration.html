    [docs]def main(raw_pol_dir, output, raw_dark_dir, threads=1):
    """Run raw data (i.e., a PolCal CS observation) through the Instrument Polarization Calibration pipeline

    The steps in the pipeline are:

    1. Generate intermediate (dark, geocal) Calibration objects associated with CS

    2. Perform sub-sample of DL-NIRSP reductions on entire CS data set

    3. Fit CU parameters and generate demodulation matrices from the resulting SoCCs

    Parameters
    ----------
    raw_pol_dir : str
        Path to directory containing raw PolCal data

    output : str
        Path were final InstrumentPolCalibration (i.e., demodulation matrices) will be saved

    raw_dark_dir : str
        Path to directory containing raw dark data associated with PolCal observations

    threads : int
        The maximum number of cpu threads to use. Note that some steps of this pipeline only use a maximum of 2 threads.

    Returns
    -------
    `Data.DemodulationData`
        The full-frame demoduation matrices. This array has shape (X, Y, 4, M) and lives in beam1 of this object
    """
    pol_rdx_dir = '{}/proc'.format(raw_pol_dir)
    inst_geo_name = '{}/proc/InstPolGeoCal.asdf'.format(raw_pol_dir)
    dmod1_name = '{}/DMOD_beam1.fits'.format(pol_rdx_dir)
    dmod2_name = '{}/DMOD_beam2.fits'.format(pol_rdx_dir)
    cmp1_name = '{}/CUpars_beam1.fits'.format(pol_rdx_dir)
    cmp2_name = '{}/CUpars_beam2.fits'.format(pol_rdx_dir)

    if os.path.exists(dmod1_name) and os.path.exists(dmod2_name):
        print('{}: Found previous demodulation matrices in {}'.format(tag(), pol_rdx_dir))
        raw_dmod = Data.DemodulationData()
        raw_dmod.load_beam_from_file(dmod1_name, 1)
        print('{}: loaded {}'.format(tag(), dmod1_name))

        raw_dmod.load_beam_from_file(dmod2_name, 2)
        print('{}: loaded {}'.format(tag(), dmod2_name))

    else:
        if os.path.exists(cmp1_name) and os.path.exists(cmp2_name):
            print('{}: Found previous fit to CU parameters in {}'.format(tag(), pol_rdx_dir))

        else:
            if not os.path.exists('{}/rdx_complete'.format(pol_rdx_dir)):
                pol_args = {'raw_pol_dir': raw_pol_dir,
                            'raw_dark_dir': raw_dark_dir,
                            'threads': threads}
                print('{}: Performing DL-NIRSP reduction on CS data in {}'.format(tag(), raw_pol_dir))
                reduce_full_CS(**pol_args)

            print('{}: Using PA&C Modules to fit CU parameters'.format(tag()))
            run_PAC_CUpar(pol_rdx_dir, threads=threads)

        print('{}: Using PA&C Modules to compute demodulation matrices'.format(tag()))
        raw_dmod = run_PAC_demod(pol_rdx_dir, threads=threads)

    if not os.path.exists(inst_geo_name):
        raise FileNotFoundError("Could not find {}. This really shouldn't have happened".format(inst_geo_name))

    with asdf.open(inst_geo_name) as f:
        inst_geo_cal = f.tree

        print('{}: Generating Instrument Polarimetric Calibration'.format(tag()))
        instpol_calibration = generate_instrument_pol_calibration(raw_dmod, inst_geo_cal, output)

    return instpol_calibration
    



    [docs]def reduce_full_CS(raw_pol_dir, raw_dark_dir, threads=1,
                   dark_cal='DarkCal.fits', geo_cal='InstPolGeoCal.asdf'):
    """Spawn the parallel reduction of an entire Calibration Sequence.

    This function sets up the right directories and default file names and then hands off the management to
    `manage_CS_rdx`.
    """
    outdir = '{}/proc'.format(raw_pol_dir)
    os.makedirs(outdir, exist_ok=True)

    darkcal_file = '{}/{}'.format(outdir, dark_cal)
    geocal_file = '{}/{}'.format(outdir, geo_cal)

    pol_kwargs = gen_inst_pol_cal_objects(raw_dark_dir, darkcal_file, raw_pol_dir, geocal_file)

    manage_kwargs = {'raw_pol_dir': raw_pol_dir,
                     'output_dir': outdir,
                     'threads': threads}

    manage_CS_rdx(manage_kwargs, pol_kwargs)

    # Create a file that indicates all files were successfully reduced
    with open('{}/rdx_complete'.format(outdir), 'w') as f:
        f.write('DL-NIRSP PolCal reduction of Calibration sequence in {}\nsaved to {}/beam[12] on {}'. \
                format(raw_pol_dir, outdir, time.asctime()))
    



    [docs]def manage_CS_rdx(manage_kwargs, pol_kwargs):
    """Manage the parallel reduction of an entire Calibration Sequence.
    
    This function constructs a separate job for each CS step and then runs them such that there are never more jobs
    running than the requested number of threads.
    """
    numthread = int(manage_kwargs['threads'])
    print('{}: Running DL-NIRSP reduction on all files with {} threads'.format(tag(), numthread))
    print('{}: Scanning polcal files in {}'.format(tag(), manage_kwargs['raw_pol_dir']))
    SoCCData = Data.SoCCData(manage_kwargs['raw_pol_dir'])
    num_steps = SoCCData.num_CS_steps
    print('{}: Found {} calibration sequence steps'.format(tag(), num_steps))

    # This loop sets up a list of mp processes. They have not been started.
    jobs = []
    pol_kwargs['dark_lock'] = mp.Lock()
    outpre = manage_kwargs['output_dir']
    for j in range(num_steps):
        jobs.append(mp.Process(target=reduce_single_step,
                               args=[SoCCData, j, outpre],
                               kwargs=pol_kwargs,
                               name='step_{:02}'.format(j)))

    # Now, start jobs and keep starting jobs until the max number of threads is reached
    #  Make a check every second to see if any jobs have finished and thus created room for new ones to start.
    i = 0
    while i < len(jobs):
        while [x.is_alive() for x in jobs].count(True) < numthread:
            jobs[i].start()
            i += 1
            if i >= len(jobs):
                break
        time.sleep(1)

    # Finally, make sure we wait for all jobs to complete before moving on.
    for p in jobs:
        p.join()

    return
    



    [docs]def gen_inst_pol_cal_objects(raw_dark_dir, dark_cal, raw_pol_dir, geo_cal):
    """Generate the Dark and Geometric Calibration objects used to reduce a PolCal Calibration Sequence.

    Note that PolCal uses a simplified version of the Geometric Calibration algorithm used for science data. Because we
    are not allowed to resample the PolCal data in any way, the Geometric Calibration only defines pixel borders that
    can be used to slice the DL-NIRSP slitbeams. This relies on the camera being well aligned with the dispersion
    direction.
    """
    # Dark
    ######
    if not os.path.exists(dark_cal):
        print('{}: Could not find {}, attempting to generate with DarkCalibration'.format(tag(), dark_cal))
        DarkCal, DarkSig, _ = DarkCalibration.main(raw_dark_dir, DLNIRSPPipeline.strip_fits(dark_cal))
    else:
        DarkCal = Data.FitsData(dark_cal)
        print('{}: Loaded {}'.format(tag(), dark_cal))

    # Inst Pol Geo Cal
    ##################
    if not os.path.exists(geo_cal):
        print('{}: Could not find {}, attempting to generate with gen_inst_pol_geo_cal'.format(tag(), geo_cal))
        tmpDrawer = Data.SoCCData(raw_pol_dir)
        tmpCC = tmpDrawer.get_CC_data(0)
        GeoCal = gen_inst_pol_geo_cal(tmpCC, geo_cal)
        borders = GeoCal['BORDERS']['data']
    else:
        with asdf.open(geo_cal) as f:
            GeoCal = f.tree
            borders = np.array(GeoCal['BORDERS']['data'])
        print('{}: Loaded {}'.format(tag(), geo_cal))

    pol_kwargs = {'borders': borders,
                  'dark_cal': DarkCal}

    return pol_kwargs
    


    [docs]def reduce_single_step(SoCCData, CS_step, outdir, borders, dark_cal=None, dark_lock=None):
    """Reduce a single CS step

    This basic reduction only applies the Dark Calibration and then chops the slitbeams apart from each other. Finally,
    each slitbeam is averaged into a single point for analysis by the PAC Modules.
    """
    polData = SoCCData.get_CC_data(CS_step)
    print('{}: Loaded CS step {} from {}'.format(tag(), CS_step, SoCCData))

    if dark_cal is not None:
        print('{}: doing dark correction'.format(tag()))
        with dark_lock:
            print('{}: acquring Dark multiprocessing lock'.format(tag()))
            generic.dark_correction(polData, dark_cal, polData, dark_cal) # Placeholder error stuff for now
            print('{}: releasing Dark multiprocessing lock'.format(tag()))

    avgData = generic.average_modulated_FitsData(polData)
    print('{}: extracting macropixels'.format(tag()))
    binData1, binData2 = extract_macropixels(avgData, borders)

    write_reduced_CS_step(binData1, binData2, outdir, CS_step)

    return binData1, binData2
    


    [docs]def extract_macropixels(fitsData, borders):
    """Extract macropixels and conform data for use by the PAC Modules

    Currently this function simply averages each slitbeam into a single pixel. Because PAC expects 3D data the final
    form for each beam is (1, 1, S), where S is the number of DL slits.

    In the future more advanced macropixilization will result in data with an arbitrary shape (X, Y, S).

    Parameters
    ----------
    fitsData : `Data.FitsData`
        Data to extract macropixels from. Should be a single, reduced CS step.

    borders : `numpy.ndarray`
        Pixel indices of the borders between each slitbeam

    Returns
    -------
    `Data.FitsData`
        Data from Beam 1 suitable for use by the PAC Modules

    `Data.FitsData`
        Data from Beam 2 suitable for use by the PAC Modules
    """
    nummod = fitsData.headers[0]['DLN__014']

    beam1_hdulist = [pyfits.PrimaryHDU(header=fitsData.headers[0])]
    beam2_hdulist = [pyfits.PrimaryHDU(header=fitsData.headers[0])]
    for i in range(nummod):
        idx = fitsData.find_data_keyword_idx('DLN__015', i)[0]
        slitbeam_list = SolarFlatCalibration.extract(fitsData.data[idx], borders)
        beam1 = np.dstack([np.nanmedian(s)[None, None] for s in slitbeam_list[::2]])
        beam2 = np.dstack([np.nanmedian(s)[None, None] for s in slitbeam_list[1::2]])

        beam1_hdulist.append(pyfits.ImageHDU(beam1, header=fitsData.data_hdus[idx].header))
        beam2_hdulist.append(pyfits.ImageHDU(beam2, header=fitsData.data_hdus[idx].header))

    binData1 = Data.FitsData(allow3D=True)
    binData1.hdulist = beam1_hdulist
    binData2 = Data.FitsData(allow3D=True)
    binData2.hdulist = beam2_hdulist

    return binData1, binData2

    """Simply handles the default directory/file naming for saving a single CS step
    """
    os.makedirs('{}/beam1'.format(outdir), exist_ok=True)
    os.makedirs('{}/beam2'.format(outdir), exist_ok=True)

    for obj, post, beamnum in zip([binData1, binData2],
                                  ['.fits', '.fits'],
                                  [1, 2]):
        if obj is not None:
            outname = '{0}/beam{1}/DLN_CS_{2:02}.PAC.beam{1}{3}'.format(outdir, beamnum, CS_step, post)
            obj.write_out(outname, overwrite=True)

    pass
    


    [docs]def gen_inst_pol_geo_cal(fitsData, output, debug=False):
    """Compute a "Geometric Calibration" for use in reducing PolCal data.

    In reality this Calibration will just be the pixel indices of the slitbeam borders and a binary mask used to
    identify the good data regions on the chip.

    This function makes use of individual helper functions from `SolarFlatCalibration`
    """
    avgData = generic.average_FitsData(fitsData) # This assumes essentially no offsets between modulator states
    binary = SolarFlatCalibration.make_binary(avgData.data[0], numotsu=1, debug=debug)

    borders = SolarFlatCalibration.find_borders(binary, 'vertical', debug=debug)

    geo_cal = {'MASK': {'Comment': 'Binary mask for un-resampled PolCal Data',
                        'data': binary.astype(bool)},
               'BORDERS': {'Comment': 'Pixel indices of slitbeam borders',
                           'data': borders}}

    asdf.AsdfFile(geo_cal).write_to(output)

    return geo_cal
    


    [docs]def run_PAC_CUpar(run_dir, threads=1, overwrite=False):
    """Take a DL-NIRSP reduced set of Calibration Sequence data and compute a fit to the CU parameters

    See the PACPipeline for more details.

    Parameters
    ----------
    run_dir : str
        Location of the processed, macropixel'd DL data associated with a single Calibration Sequence (CS).

    threads : int
        Maximum number of cpus to use when fitting CU parameters. No more than 2 cpus will ever be used.

    overwrite : bool
        If False then demodulation matrices will not be computed if they already exist.
    """
    from PAC_Pipeline import FitCUParams
    from PAC_Pipeline import generic as PAC_generic
    telescope_db = PAC_generic.get_default_telescope_db()

    jobs = []

    pac_kwargs = {'overwrite': overwrite,
                  'save_stats': True,
                  'telescope_db': telescope_db}
    for i in range(1, 3):
        beam_dir = '{}/beam{}'.format(run_dir, i)
        beam_output = '{}/CUpars_beam{}.fits'.format(run_dir, i)
        jobs.append(mp.Process(target=FitCUParams.main,
                               args=[beam_dir, beam_output],
                               kwargs=pac_kwargs,
                               name='Beam{}'.format(i)))

    i = 0
    while i < len(jobs):
        while [x.is_alive() for x in jobs].count(True) < threads:
            jobs[i].start()
            i += 1
            if i >= len(jobs):
                break
        time.sleep(1)

    for p in jobs:
        p.join()

    return
    


    [docs]def run_PAC_demod(run_dir, threads=1, overwrite=False):
    """Take a DL-NIRSP reduced set of Calibration Sequence data and compute a set of demodulation matrices.

    The best demodulation is found using pre-computed CU parameters like those generated by `run_PAC_CU_parameters`

    Parameters
    ----------
    run_dir : str
        Location of the processed, macropixel'd ViSP data associated with a single Calibration Sequence (CS).

    threads : int
        Maximum number of cpus to use when fitting CU parameters. No more than 2 cpus will ever be used.

    overwrite : bool
        If False then demodulation matrices will not be computed if they already exist.

    Returns
    -------
    Data.DemodulationData
        Object containing the demodulation matrices for both ViSP beams
    """
    from PAC_Pipeline import GenerateDemodMatrices
    from PAC_Pipeline import generic as PAC_generic

    jobs = []

    pac_kwargs = {'overwrite': overwrite,
                  'telescope_db': PAC_generic.get_default_telescope_db()}
    for i in range(1, 3):
        beam_dir = '{}/beam{}'.format(run_dir, i)
        CUpar_file = '{}/CUpars_beam{}.fits'.format(run_dir, i)
        beam_output = '{}/DMOD_beam{}.fits'.format(run_dir, i)
        jobs.append(mp.Process(target=GenerateDemodMatrices.main,
                               args=[beam_dir, CUpar_file, beam_output],
                               kwargs=pac_kwargs,
                               name='Beam{}'.format(i)))

    i = 0
    while i < len(jobs):
        while [x.is_alive() for x in jobs].count(True) < threads:
            jobs[i].start()
            i += 1
            if i >= len(jobs):
                break
        time.sleep(1)

    for p in jobs:
        p.join()

    raw_dmod = Data.DemodulationData()
    raw_dmod.load_beam_from_file('{}/DMOD_beam1.fits'.format(run_dir), 1)
    raw_dmod.load_beam_from_file('{}/DMOD_beam2.fits'.format(run_dir), 2)

    return raw_dmod
    


    [docs]def generate_instrument_pol_calibration(dmodData, geo_cal, output):
    """Read raw demodulation matrices and prep them for use in Science data reduction

    The demodulation matrices for each DL-NIRSP beam are treated separately and if either one was a full-frame average
    then the spatial dimensions are simply squeezed out. Otherwise the data are passed to `upsample_demodulation` for
    interpolation to the full beam size.

    Parameters
    ----------
    dmodData : Data.DemodulationData
        The demodulation matrix for both DL beams. Should have a single data HDU for each beam with shape
        (x, y, l, 4, M) where (x, y, l) is the shape of the binned DL-NIRSP beams frame passed to PA&C and M is the
        number of unique modulator states.

    geo_cal : dict
        Polarimetric geocal object generated by ``gen_inst_pol_geo_cal``. The `mask` from this is used to mask out the
        bad regions in the final demodulation matrices.

    output : str
        Location to save the final demodulation matrices

    Returns
    -------
    Data.DemodulationData
        Object containing a data HDU for each DL-NIRSP beam. The data shape of these HDUs will either be (X, Y, L, 4, M)
        or (4, M), where X and Y are the full read-out size of a single DL-NIRSP beam and L is the number of slits (5).
    """
    print('{}: extracting demoulation matrices'.format(tag()))
    numx1, numy1, numl1, _, numm1 = dmodData.beam1.data.shape
    numx2, numy2, numl2, _, numm2 = dmodData.beam2.data.shape

    if numm1 != numm2:
        raise ValueError("The demodulation matrices don't have the same number of modulator states")

    if not (numx1 == numx2 == numy1 == numy2 == 1):
        raise ValueError("This pipeline currently only supports full slit averages")

    borders = geo_cal['BORDERS']['data']
    OG_shape = geo_cal['MASK']['data'].shape
    big_dmod = np.zeros(OG_shape + (4, numm1), dtype=np.float32)
    for i in range(borders.size - 1)[::2]:
        print('{}: filling [:, {}:{}, :, :] with beam 1 [0, 0, {}, :, :]'.format(tag(), borders[i], borders[i+1], i//2))
        big_dmod[:, borders[i]:borders[i+1], :, :] = dmodData.beam1.data[0, 0, i//2, :, :]
    for i in range(borders.size - 1)[1::2]:
        print('{}: filling [:, {}:{}, :, :] with beam 2 [0, 0, {}, :, :]'.format(tag(), borders[i], borders[i+1], i//2))
        big_dmod[:, borders[i]:borders[i+1], :, :] = dmodData.beam1.data[0, 0, i//2, :, :]

    print('{}: applying mask'.format(tag()))
    mask = geo_cal['MASK']['data']
    for i in range(4):
        for j in range(numm1):
            big_dmod[:, :, i, j][mask] = np.nan

    final_dmod = Data.DemodulationData()
    final_dmod.init_beam_with_ndarray(big_dmod, 1) # Kind of a hack for now. Maybe not a hack at all.

    final_dmod.write_out(output, overwrite=True)

    return final_dmod
