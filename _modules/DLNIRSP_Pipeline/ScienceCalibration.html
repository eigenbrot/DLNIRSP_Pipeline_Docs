    [docs]def main(datasetData, MRN, outpre, dark_cal=None, instrument_pol_cal=None, solar_flat_cal=None,
         geometric_cal=None, dark_cal_sig=None, solar_flat_cal_sig=None, allow_no_pol=False,
         ifu_cal=None, cube_shape=(100, 100), telescope_db=None, remove_TM=True, include_M12=True,
         remove_parallactic=True):
    """Reduce DLNIRSP Science data

    Any intermediate Calibration objects that are provided will be applied to the data and are technically optional.
    If no Calibration objects are provided then the data will essentially be passed through, which is pretty boring.
    An slight exception to this is the ``instrument_pol_cal``; this is psuedo-required and a user must prove their
    sanity via the ``allow_no_pol`` keyword to turn off demodulation.

    Parameters
    ----------
    datasetData : `Data.DataSetData`
        Object containing mapping of mosaic/repeat numbers to raw data files

    MRN : `Data.MosaicRepeatNumber`
        Identified of the mosaic/repeat step to reduce

    outpre : str
        Prefix of filename to save reduced Science object

    dark_cal : `Data.FitsData`
        Dark Calibration object

    instrument_pol_cal : `Data.DemodulationData`
        Instrument Polarization Calibration object (i.e., demodulation matrices)

    solar_flat_cal : `Data.FitsData`
        Solar Flat Calibration object

    geometric_cal : dict
        Geometric Calibration object

    dark_cal_sig : `Data.FitsData`
        Uncertainty associated with ``dark_cal``

    solar_flat_cal_sig : `Data.FitsData`
        Uncertainty associated with ``solar_flat_cal``

    allow_no_pol : bool
        If True then let the script continue without ``instrument_pol_cal``

    ifu_cal : dict
        Dictionary containing IFU remapping information

    cube_shape : tuple
        (X, Y) tuple that defines the final spatial dimensions (in pixels) of the resampled IFU data cube

    Returns
    -------
    `Data.FitsData`
        Reduced science data. If IFU resampling is done (i.e., Level 1.5 data) there will be four 3D data_hdus,
        corresponding to the four Stokes vectors. If no ``ifu_cal`` is provided then the output will be Level 1 data,
        with each of the four data_hdus containing all DL fibers stacked into a single slit.
    """
    print('{}: Loading DL-NIRSP data for {} from {}'.format(tag(), MRN, datasetData))
    sciData = datasetData.get_MR_data(MRN)

    validate_observe_task_data(sciData)

    generic.linearity_correction(sciData)

    if dark_cal is not None:
        if dark_cal_sig is None:
            raise ValueError('Dark Calibration specified without associated sigma object')
        print('{}: doing Dark Correction'.format(tag()))

        ## This is a stop-gap until I hear from SJ about how to really handle uncertainty/sigma frames
        from DLNIRSP_Pipeline.utils import gen_fake_data
        data_sig = gen_fake_data.get_fake_sig(sciData)
        ####

        generic.dark_correction(sciData, dark_cal, data_sig, dark_cal_sig)

    if instrument_pol_cal is not None:
        avgData = generic.average_modulated_FitsData(sciData) # Will this ever do anything??

        print('{}: applying polarization calibration'.format(tag()))
        stokesData = generic.polarization_correction(avgData, instrument_pol_cal,
                                                     remove_TM=remove_TM, telescope_db=telescope_db,
                                                     include_M12=include_M12, remove_parallactic=remove_parallactic)
    elif allow_no_pol:
        stokesData = sciData
    else:
        raise ValueError('No Instrument Polarization Calibration provided! If you really want to do this use `allow_no_pol`')

    if solar_flat_cal is not None:
        if solar_flat_cal_sig is None:
            raise ValueError('SolarGain Calibration specified without associated sigma object')
        print('{}: doing Solar Gain Correction'.format(tag()))

        ## This is a stop-gap until I hear from SJ about how to really handle uncertainty/sigma frames
        from DLNIRSP_Pipeline.utils import gen_fake_data
        data_sig = gen_fake_data.get_fake_sig(sciData)
        ####

        generic.demodulated_gain_correction(stokesData, solar_flat_cal, data_sig, solar_flat_cal_sig)

    if geometric_cal is not None:
        print('{}: doing geometric correction'.format(tag()))
        stokesData = generic.geometric_correction(stokesData, geometric_cal)

    if ifu_cal is not None:
        if geometric_cal is None:
            raise ValueError('Cannot perform IFU resampling without Geometric Calibration')

        print('{}: Applying IFU resampling'.format(tag()))
        cubeData = generic.IFU_correction(stokesData, ifu_cal, cube_shape)

    else:
        cubeData = stokesData

    sci_cal = generate_science_calibration(cubeData, outpre, MRN)

    return sci_cal

    [docs]def generate_science_calibration(fitsData, outpre, MRN):
    """Generate a sensible file name and save to FITS.

    The filename will be based on the mosiac/repeat number of the object in question and will be:

               "outpre_MRN.fits"

    Parameters
    ----------
    fitsData : `Data.FitsData`
        The object to be saved

    outpre : str
        A string to prefix on the mosaic/repeat number designation

    MRN : `Data.MosaicRepeatNumber`
        The mosaic/repeat number object for the given data

    Returns
    -------
    `Data.FitsData`
        Passthrough of ``fitsData``
    """
    outname = '{}_{}.{}'.format(outpre, MRN.get_str(), CONSTANTS['Data']['suffix'])
    print('{}: Saving data to {}'.format(tag(), outname))
    fitsData.write_out(outname, overwrite=True)

    return fitsData
