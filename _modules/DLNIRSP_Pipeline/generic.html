    [docs]def average_FitsData(fitsData):
    """Average ALL hdus into a single frame

    The DATE header keywords are updated to be the average of all used frames

    Parameters
    ----------
    fitsData : `Data.FitsData`
        Data to be averaged

    Returns
    -------
    `Data.FitsData`
        Will have a single data_hdu that contains the average of all input data_hdus
    """
    numx, numy = fitsData.data[0].shape
    hdulist = [pyfits.PrimaryHDU(header=fitsData.headers[0])]

    print('{}: Averaging all HDUs from {}'.format(tag(), fitsData))
    output = np.zeros((numx, numy), dtype=np.float32)
    start_times = []
    end_times = []
    for i, h in enumerate(fitsData.data_hdus):
        output += fitsData.data[i]
        start_times.append(Time(h.header['DATE-BGN']))
        end_times.append(Time(h.header['DATE-END']))
    hdulist.append(pyfits.ImageHDU(output / len(fitsData.data_hdus), fitsData.headers[1]))

    new_start_time = min(start_times)
    new_end_time = max(end_times)
    for h in hdulist:
        h.header['DATE-BGN'] = new_start_time.fits
        h.header['DATE-END'] = new_end_time.fits
        h.header['DATE-OBS'] = (new_start_time + (new_end_time - new_start_time) / 2).fits
    newData = Data.FitsData()
    newData.hdulist = hdulist
    newData.origin = 'average_FitsData'

    return newData
    


    [docs]def average_modulated_FitsData(fitsData):
    """Average all data from the same modulator state together

    The DATE header keywords are updated to be the average of all the frames corresponding to a particular modulator
    state

    Parameters
    ----------
    fitsData : `Data.FitsData`
        Data to be averaged together. Different modulator states are identified via their SPEC-0122 headers.

    Returns
    -------
    `Data.FitsData`
        The average of each modulator state will live in a separate data_hdu
    """
    numx, numy = fitsData.data[0].shape
    hdulist = [pyfits.PrimaryHDU(header=fitsData.headers[0])]
    nummod = fitsData.headers[0]['DLN__014']

    for i in range(nummod):
        pos_idx = fitsData.find_data_keyword_idx('DLN__015', i)
        if len(pos_idx) > 0:
            output = np.zeros((numx, numy), dtype=np.float32)
            start_times = []
            end_times = []
            print('{}: Averaging modulator state {}'.format(tag(), i))
            for j in pos_idx:
                print('{}:  using HDU {}'.format(tag(), j))
                output += fitsData.data[j]
                start_times.append(Time(fitsData.data_hdus[j].header['DATE-BGN']))
                end_times.append(Time(fitsData.data_hdus[j].header['DATE-END']))
            hdulist.append(pyfits.ImageHDU(output / len(pos_idx), header=fitsData.data_hdus[pos_idx[0]].header))
            new_start_time = min(start_times)
            new_end_time = max(end_times)
            hdulist[-1].header['DATE-BGN'] = new_start_time.fits
            hdulist[-1].header['DATE-END'] = new_end_time.fits
            hdulist[-1].header['DATE-OBS'] = (new_start_time + (new_end_time - new_start_time) / 2).fits

    newData = Data.FitsData()
    newData.hdulist = hdulist
    newData.origin = 'average_modulated_FitsData'

    return newData
    


    [docs]def stddev_FitsData(fitsData, average=None):
    """Compute the per-pixel standard deviation across ALL HDUs present in the input data

    The DATE header keywords are updated to be the average of all data frames used

    Parameters
    ----------
    fitsData : `Data.FitsData`
        Data to compute the stddev of

    average : `Data.FitsData`
        A precomputed total average of the input data. Almost certainly the output of `average_FitsData`. If no average
        is provided then `average_FitsData` is used to compute it.

    Returns
    -------
    `Data.FitsData`
        The standard deviation across all frames in the input data. This object will have a single data_hdu.
    """
    if average is None:
        average = average_FitsData(fitsData)

    numx, numy = fitsData.data[0].shape
    hdulist = [pyfits.PrimaryHDU(header=fitsData.headers[0])]
    avg_data = average.data[0]

    print('{}: Computing sigma on all HDUs from {}'.format(tag(), fitsData))
    output = np.zeros((numx, numy), dtype=np.float32)
    start_times = []
    end_times = []
    for i, h in enumerate(fitsData.data_hdus):
        output += (fitsData.data[i] - avg_data)**2
        start_times.append(Time(h.header['DATE-BGN']))
        end_times.append(Time(h.header['DATE-END']))
    hdulist.append(pyfits.ImageHDU(np.sqrt(output / len(fitsData.data_hdus)), fitsData.headers[1]))

    new_start_time = min(start_times)
    new_end_time = max(end_times)
    for h in hdulist:
        h.header['DATE-BGN'] = new_start_time.fits
        h.header['DATE-END'] = new_end_time.fits
        h.header['DATE-OBS'] = (new_start_time + (new_end_time - new_start_time) / 2).fits
    newData = Data.FitsData()
    newData.hdulist = hdulist
    newData.origin = 'stddev_FitsData'

    return newData

            start_times = []
            end_times = []
            avg_idx = average.find_data_keyword_idx('DLN__015',i)

    """Compute the full-frame average for each data_hdu present in the input data.

    Parameters
    ----------
    fitsData : `Data.FitsData`
        Should probably have more than one data_hdu

    Returns
    -------
    `numpy.ndarray`

    [docs]def dark_correction(fitsData, dark_calibration, data_sig, dark_cal_sig):
    """Subtract a dark frame from all data_hdus in the input data

    Also propagate the uncertainties

    Parameters
    ----------
    fitsData : `Data.FitsData`
        Data to perform dark subtraction on

    dark_calibration : `Data.FitsData`
        Object containing the dark frame to use. If more than one data_hdu are present, only the first will be used.

    data_sig : `Data.FitsData`
        Uncertainty associated with ``fitsData``

    dark_cal_sig : `Data.FitsData`
        Uncertainty associated with ``dark_calibration``

    Returns
    -------
    Nothing, the correction is applied in place
    """
    print('{}: Performing dark correction on {} with {}'.format(tag(), fitsData, dark_calibration))
    dark = dark_calibration.data[0]
    dark_sig = dark_cal_sig.data[0]

    for i in range(len(fitsData.data)):
        print('{}: subtracting dark from HDU {}'.format(tag(), i))
        fitsData.data[i] = fitsData.data[i] - dark
        data_sig.data[i] = np.sqrt(dark_sig**2 + data_sig.data[i]**2)

    return
    


    [docs]def modulated_gain_correction(fitsData, gain_calibration, data_sig, gain_cal_sig):
    """Perform gain correction on modulated data

    In other words, divide each modulator state in the input data by the gain from the same modulator state.
    Uncertainties are propagated.

    Parameters
    ----------
    fitsData : `Data.FitsData`
        Data to correct

    gain_calibration : `Data.FitsData`
        Gain frames to use. Should have the same number of modulator states as ``fitsData``

    data_sig : `Data.FitsData`
        Uncertainty associated with ``fitsData``

    gain_cal_sig : `Data.FitsData`
        Uncertainty associated with ``gain_calibration``

    Returns
    -------
    Nothing, the correction is done in place
    """
    print('{}: Performing gain correction on {} with {}'.format(tag(), fitsData, gain_calibration))
    nummod = fitsData.headers[0]['DLN__014']

    try:
        if fitsData.headers[0]['ISSTOKES']:
            raise ValueError('You are trying to correct demodulated data. Use generic.demodulated_gain_correction instead')
    except KeyError:
        pass
    try:
        if gain_calibration.headers[0]['ISSTOKES']:
            raise ValueError('You are trying to correct demodulated data with a modulated Calibration. '
                             'Use generic.demodulated_gain_correction instead')
    except KeyError:
        pass

    for i in range(nummod):
        print('{}: modulator state {}'.format(tag(), i))
        didx = fitsData.find_data_keyword_idx('DLN__015', i)
        gidx = gain_calibration.find_data_keyword_idx('DLN__015', i)
        if len(gidx) == 0:
            raise ValueError('Could not find Gain data for modulator state {}'.format(i))
        if len(gidx) > 1:
            print('{}: WARNING. Gain object {} has more than one HDU for modulator {}. This is weird.'.format(tag(),
                                                                                                              gain_calibration,
                                                                                                              i))
        gain = gain_calibration.data[gidx[0]]
        gain_sig = gain_cal_sig.data[gidx[0]]
        for j in didx:
            print('{}:  propogating uncertainty in HDU {}'.format(tag(), j))
            data_sig.data[j] = np.sqrt((data_sig.data[j] / gain)**2 + (gain_sig * fitsData.data[j] / gain**2)**2)
            print('{}:  dividing from HDU {}'.format(tag(), j))
            fitsData.data[j] = fitsData.data[j] / gain

    return
    


    [docs]def demodulated_gain_correction(fitsData, gain_calibration, data_sig, gain_cal_sig):
    """Divide each input Stokes state by Stokes I of the gain data

    Uncertainties are propagated

    Parameters
    ----------
    fitsData : `Data.FitsData`
        Demodulated data to do correction on. Needs to have 4 data_hdus

    gain_calibration : `Data.FitsData`
        Demodulated gain data. Only the Stokes I HDU will be used.

    data_sig : `Data.FitsData`
        Uncertainty associated with ``fitsData``

    gain_cal_sig : `Data.FitsData`
        Uncertainty associated with ``gain_calibration``

    Returns
    -------
    Nothing. The correction is done in place.
    """
    print('{}: Performing gain correction on {} with {}'.format(tag(), fitsData, gain_calibration))

    if 'ISSTOKES' not in fitsData.headers[0].keys() or not fitsData.headers[0]['ISSTOKES']:
        raise ValueError('You are trying to correct modulated data. Use generic.modulated_gain_correction instead')

    if 'ISSTOKES' not in gain_calibration.headers[0].keys() or not gain_calibration.headers[0]['ISSTOKES']:
        raise ValueError('You are trying to correct demodulated data with a modulated Calibration. '
                         'Use generic.modulated_gain_correction instead')

    gain_data = gain_calibration.data[0] # Stokes I
    gain_sig = gain_cal_sig.data[0]
    for i in range(4):
        print('{}: stokes component {}'.format(tag(), i))
        print('{}:  propogating uncertainty'.format(tag()))
        data_sig.data[i] = np.sqrt((data_sig.data[i] / gain_data)**2 + (gain_sig * fitsData.data[i] / gain_data**2)**2)
        print('{}:  dividing gain'.format(tag()))
        fitsData.data[i] = fitsData.data[i] / gain_data

    return
    


    [docs]def polarization_correction(fitsData, inst_pol_calibration,
                            remove_TM=True, telescope_db=None, include_M12=True, remove_parallactic=True):
    """Remove some or all polarimetric effects from data

    At a minimum, this function demodulates the data with an InstrumentPolarizationCalibration object (i.e., a
    demodulation matrix). It can also remove the M1 - M6 telescope Mueller matrix. In this case it first uses data
    header values to generate a telescope matrix appropriate for the time and date of observations and then removes
    this matrix from the data.

    Parameters
    ----------
    fitsData : `Data.FitsData`
        Data to remove polarimetric effects from. Should be modulated data.

    inst_pol_calibration : `Data.DemodulationData`
        The demodulation matrices

    remove_TM : bool
        If True then the inverse telescope Mueller matrix is applied

    telescope_db : str
        Path to a file containing telescope mirror parameters as a function of time and wavelength. If this is None then
        the default database is loaded from the PA&C Modules (recommended).

    include_M12 : bool
        If True then mirrors M1 and M2 are included in the telescope Mueller matrix

    remove_parallactic : bool
        If True then the telescope Mueller matrix will include rotation by the parallactic angle and will thus put
        the data into a common solar reference frame

    Returns
    -------
    `Data.FitsData`
        The demodulated input data with the telescope Mueller matrix potentially removed
    """
    print('{}: demodulating data'.format(tag()))
    stokesData = instrument_polarization_correction(fitsData, inst_pol_calibration)

    if remove_TM:
        if telescope_db is None or not os.path.exists(telescope_db):
            print('{}: Loading telescope parameter database from PA&C defualts'.format(tag()))
            from PAC_Pipeline.generic import get_default_telescope_db
            telescope_db = get_default_telescope_db()

        print('{}: generating inverse telescope matrix'.format(tag()))
        tm_mat = TelescopePolarizationCalibration.generate_telescope_pol_calibration(stokesData, telescope_db,
                                                                                     M12=include_M12,
                                                                                     include_parallactic=remove_parallactic)
        print('{}: removing telescope matrix'.format(tag()))
        stokesData = telescope_pol_correction(stokesData, tm_mat)
    else:
        print('{}: skipping removal of telescope matrix'.format(tag()))

    return stokesData
    


    [docs]def instrument_polarization_correction(fitsData, inst_pol_calibration):
    """Multiply input data by the provided demodulation matrix

    Parameters
    ----------
    fitsData : `Data.FitsData`
        Input data. Should have M data_hdus (one for each modulator state), each with shape (x, y).

    inst_pol_calibration : `Data.Demodulation`
        Should contain the full-frame (i.e., both beams) demodulation matrices in the `.beam1` property. These matrices
        should have shape (x, y, 4, M) or (4, M).

    Returns
    -------
    `Data.FitsData`
        Demodulated Stokes data. Has four data_hdus each of shape (x, y), corresponding to the four Stokes components.
    """
    print('{}: stacking data'.format(tag()))
    stacked_data = da.dstack(fitsData.data)

    # Remember from InstrumentPolarizationCalibration.generate_instrument_pol_calibration that beam1 contains
    #  the full-frame (both beam) demodulation matrices
    demod = da.from_array(inst_pol_calibration.beam1.data)

    if stacked_data.shape[2] != demod.shape[3]:
        raise ValueError('Modulated data ({}) and demodulation matrices ({}) do not '
                         'have the same number of modulation states'.format(stacked_data.shape[2], demod.shape[3]))

    output = polarimetric_multiplication(stacked_data, demod)

    if is_mp():
        print('{}: subprocess detected. Computing array synchronously'.format(tag()))
        with dask.config.set(scheduler='synchronous'):
            output = output.compute()
    else:
        print('{}: computing array (potentially) asynchronously'.format(tag()))
        output = output.compute()

    del stacked_data

    outData = Data.FitsData()
    outData.init_with_ndarray(output)
    outData.hdulist[0].header = fitsData.hdulist[0].header.copy()
    outData.hdulist[0].header['ISSTOKES'] = True

    for h, stokes in zip(outData.data_hdus, ['I', 'Q', 'U', 'V']):
        h.header = outData.hdulist[0].header.copy()
        h.header['STOKES'] = stokes
        h.header['EXTNAME'] = 'Stokes {}'.format(stokes)

    return outData
    


    [docs]def telescope_pol_correction(fitsData, teleCal):
    """Apply an inverse Telescope Mueller matrix and update header values

    Parameters
    ----------
    fitsData : Data.FitsData
        Data to apply the inverse TM to. Should have 4 data HDUs (stokes vectors).

    teleCal : numpy.ndarray

    output = polarimetric_multiplication(stacked_data, teleCal)

            output = output.compute()
    else:
        print('{}: computing array (potentially) asynchronously'.format(tag()))

    outData = Data.FitsData()
    outData.init_with_ndarray(output)
    outData.hdulist[0].header = fitsData.hdulist[0].header.copy()

    for new, old in zip(outData.data_hdus, fitsData.data_hdus):
        new.header = old.header.copy()

    return outData
    


    [docs]def polarimetric_multiplication(data, demod):
    """Matrix multiply the input data by a polarization demodulation matrix.

    This is a general-purpose function that requires good understanding of the input parameters. The important thing
    is that the demodulation matrix's last dimension is equal to the number of data HDU's in the input data.

    Parameters
    ----------
    data : `numpy.ndarray`
        Input data to demodulate. Should have shape (x, y, M).

    demod : `numpy.ndarray`
        The demodulation matrix. There are a few different shapes it can have:

            1. (x, y, 4, M), where (x, y) is the shape of a single input frame and M is the number of modulation
            states. In this case, a separate demodulation can be provided for each individual pixel. Total control!

            2. (4, M). This assumes a constant demodulation matrix across the ViSP chip

            3. (4, 4). This form is used for removing the polarizing effect of the telescope optics and requires the
            input data to already be in "Stokes" form.

        Note that in all cases the length of the second to last dimension is 4 (corresponding to the Stokes vector)
        and the length of the last dimension is equal to the number of data HDUs in the input data.

    Returns
    -------
    `numpy.ndarray`
        Demodulated/corrected polarization data. Regardless of the inputs, the output will have shape (x, y, 4),
        corresponding to the 4 Stokes parameters.
    """

    print('{}: matrix-multiplying arrays with shapes {} and {}'.format(tag(), data.shape, demod.shape))
    output = np.sum(demod * data[:, :, None, :], axis=3)

    return output
    


    [docs]def geometric_correction(fitsData, geometric_calibration):
    """Remove camera/spectrograph distortions and split into individual slits

    The algorithm works in this order:

    1. Correct camera rotation
    2. Correct for offsets between the two beams
    3. Combine beam 1 and beam 2
    4. Correct spectral geometry (i.e., curvature)

    As a consequence of the beam combination the non-data slit borders are removed in the final output.

    Parameters
    ----------
    fitsData : `Data.FitsData`
        Demodulated data to correct

    geometric_calibration : dict
        Geometric Calibration object produced by the `SolarFlatCalibration` module

    Returns
    -------
    `Data.FitsData`
        This object had four data_hdus, corresponding to the four Stokes vectors. Each data_hdu contains all DL fibers
        represented in a single slit.
    """
    if 'ISSTOKES' not in fitsData.headers[0].keys() or not fitsData.headers[0]['ISSTOKES']:
        raise ValueError('Geometric correction only works on demodulated data for now')

    final_hdu_list = [pyfits.PrimaryHDU(header=fitsData.headers[0])]
    numstokes = len(fitsData.data_hdus)
    for k in range(numstokes):
        stokes_par = fitsData.data_hdus[k].header['STOKES']
        print('{}: working on Stokes {}'.format(tag(), stokes_par))
        data = fitsData.data[k]

        slit_list = apply_geo_corr(data, geometric_calibration)
        full_slit = np.vstack(slit_list)

        final_hdu_list.append(pyfits.ImageHDU(full_slit, header=fitsData.data_hdus[k].header))

    outData = Data.FitsData()
    outData.hdulist = final_hdu_list

    return outData
    



    [docs]def apply_geo_corr(data, geometric_calibration):
    """Actually apply the geometric correction. This is a helper. See `geometric_correction` for more details.

    Parameters
    ----------
    data : `numpy.ndarray`
        Data to be corrected

    geometric_calibration : dict
        Geometric Calibration object produced by the `SolarFlatCalibration` module

    Returns
    -------
    list
        A list of `numpy.ndarray` objects, one for each DL-NIRSP slit.
    """
    print('{}: correcting camera rotation'.format(tag()))
    data_rot = SolarFlatCalibration.correct_rotation(data, geometric_calibration['ANGLES']['data'])

    print('{}: extracting slits'.format(tag()))
    slitbeam_list = SolarFlatCalibration.extract(data_rot, geometric_calibration['BORDERS']['data'])
    numslits = len(slitbeam_list) // 2

    slit_list = []
    for s in range(numslits):
        print('{}: working on slit {}'.format(tag(), s))
        slitbeam = slitbeam_list[s * 2:s * 2 + 2]  # [slit_s_beam1, slit_s_beam2]

        print('{}: combining beams'.format(tag()))
        slit = SolarFlatCalibration.extract_perfect_slit(slitbeam, geometric_calibration['BEAMOFF']['data'][:, s],
                                                         geometric_calibration['MASKS']['data'][s][0],
                                                         geometric_calibration['MASKS']['data'][s][1])

        print('{}: removing spectral geometry'.format(tag()))
        perfect_slit = SolarFlatCalibration.remove_spec_geometry(slit, geometric_calibration['SPECSHIFT']['data'][:, s])
        slit_list.append(perfect_slit)

    return slit_list
    


    [docs]def IFU_correction(fitsData, ifu_cal, shape, method='nearest'):
    """Resample the full DL-NIRSP slit onto a 2D spatial grid, resulting in a 3D data cube.

    The shape of the 2D spatial resampling is left up to the user. A good place to start is the sqrt of the total number
    of spatial positions in the DL slit.

    Parameters
    ----------
    fitsData : `Data.FitsData`
        Data to be resampled. Must be demodulated and have the geometric correction applied. Thus, if the raw frames
        have shape (x, y) and DL has N slits this object will have 4 data_hdus, each with an array of shape
        (~x/2N, ~y*N).

    ifu_cal : dict
        Information on the 2D remapping of the DL slit. Produced by `IFUCalibration.main`

    shape : tuple
        2D spatial shape of the resampled field-of-view

    method : str
        The interpolation method to use. Options are ``nearest`` (default), ``linear``, or ``cubic``. See the docs of
        `scipy.interpolate.griddata` for more information.

    Returns
    -------
    `Data.FitsData`
        Will have 4 data_hdus, each having data with shape (~x/2N, ``shape``[0], ``shape``[1])
    """
    points = np.stack((ifu_cal['x']['positions'], ifu_cal['y']['positions']), -1)
    mask = ~np.sum(np.isnan(points), axis=1).astype(bool)
    points = points[mask, :]

    output_hdus = [pyfits.PrimaryHDU(header=fitsData.headers[0])]
    numstokes = len(fitsData.data)

    numwave = fitsData.data[0].shape[1]
    cube_shape = (numwave,) + shape
    print('{}: output cube will have shape {}'.format(tag(), cube_shape))
    x_vec = np.linspace(np.nanmin(points[:, 0]), np.nanmax(points[:, 0]), shape[1])
    y_vec = np.linspace(np.nanmin(points[:, 1]), np.nanmax(points[:, 1]), shape[0])
    XX, YY = np.meshgrid(x_vec, y_vec)

    # TODO: Replace blank CTYPEn with correct values
    wcs_dict = {'WCSAXES': 3,
                'CRPIX1': 1, 'CRVAL1': x_vec.min(), 'CDELT1': np.diff(x_vec)[0], 'CTYPE1': ' ', 'CUNIT1': 'arcsec',
                'CRPIX2': 1, 'CRVAL2': y_vec.min(), 'CDELT2': np.diff(y_vec)[0], 'CTYPE2': ' ', 'CUNIT2': 'arcsec',
                'CRPIX3': 1, 'CRVAL3': 0, 'CDELT3': 0.1, 'CTYPE3': ' ', 'CUNIT3': 'nanometer'}
    output_hdus[0].header.update(wcs_dict)
    for k in range(numstokes):
        stokes_par = fitsData.data_hdus[k].header['STOKES']
        print('{}: working on Stokes {}'.format(tag(), stokes_par))
        data = fitsData.data[k][mask, :]

        output = np.zeros(cube_shape)
        for i in range(numwave):
            # print('{}:  wavelength channel {}'.format(tag(), i))
            output[i, :, :] = spi.griddata(points, data[:, i], (XX, YY), method=method)

        hdu = pyfits.ImageHDU(output, header=fitsData.data_hdus[k].header)
        hdu.header.update(wcs_dict)
        output_hdus.append(hdu)

    cubeData = Data.FitsData(allow3D=True)
    cubeData.hdulist = output_hdus

    return cubeData
