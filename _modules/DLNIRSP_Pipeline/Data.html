    [docs]class FitsData():
    """FitsData is a basically a wrapper for astropy.io.fits, with some extra methods and properties to aid the
    reduction of DKIST data.

    At its heart, this class is just a list of astropy.io.fits HDU objects.

    astropy.io.fits is used to store data and metadata because of its use of memmap to reduce the load on RAM when
    accessing data arrays.
    """

    def __init__(self, load_obj=None, memmap=None, allow3D=False):
        """Initialize an empty object and maybe load files from a directory.

        Parameters
        ----------
        load_obj : str
            If supplied, the path to a directory of FITS files that will be loaded into this object.

        memmap : bool or None
            Whether to use astropy.io.fits' memmap functionality. Set to None to let astropy figure it out for you.
        """
        # Set some default properties that other methods expect to exist
        self.origin = None
        self.is_tmp = False
        self.ref_file = ''
        self._hdulist = pyfits.HDUList()
        self.data_hdus = pyfits.HDUList()
        self._data_list = _DataList(allow3D)

        if load_obj is not None:
            if os.path.isdir(load_obj):
                self.load_from_directory(load_obj, memmap=memmap)
            elif os.path.exists(load_obj):
                self.load_from_file(load_obj, memmap=memmap)
            else:
                print('WARNING: {} does not appear to be a file or directory. Nothing loaded.'.format(load_obj))

    def __del__(self):
        """Only overloaded so that any temp FITS files can be deleted."""
        if self.is_tmp:
            print('Closing {} and deleting associated tmp file.'.format(self))
            os.remove(self.ref_file)

    def __repr__(self):
        if self.origin is not None:
            return 'FitsData created by {}'.format(self.origin)
        else:
            return 'FitsData loaded from {}'.format(self.ref_file)


    [docs]    def load_from_directory(self, directory, searchstr='*.fits', max_open=1000, memmap=None):
        """Load a series of FITS files from a directory.

        Each separate file will become a single HDU in this object's data_hdus. A blank PrimaryHDU is provided.

        Parameters
        ----------
        directory : str
            The path to the directory to look in

        searchstr : str
            A string used to narrow down the files loaded. For example, to search for all .fits files starting with
            "image_1", the search string would be "image_1*.fits".

        max_open : int
            The maximum number of open files to keep. Some systems will cause massive errors if too many files are
            open at a time. If loading from a directory with a lot of files, this method will flush() after max_open
            have been loaded.

        memmap : boolean or None
            Use astropy.io.fits' memmap functionality? If None then astropy will make this choice for you.
        """
        print('{}: Loading files from {}'.format(tag(), directory))
        hdulist = [pyfits.PrimaryHDU()]
        file_list = sorted(glob('{}/{}'.format(directory, searchstr)))

        if len(file_list) == 0:
            raise FileNotFoundError('Did not find any files to load in {}'.format(directory))

        i = 0
        for f in file_list:
            h = pyfits.open(f, memmap=memmap)
            if i > max_open:
                self.hdulist = hdulist
                self.flush()
                i = 0
            print('{}: Loading file {}'.format(tag(), f))
            hdulist.append(h[0])
            i += 1

        hdulist[0].header = hdulist[1].header.copy()
        self.hdulist = pyfits.HDUList(hdulist)
        self.origin = '{}/{}'.format(directory, searchstr)

    [docs]    def init_with_ndarray(self, array, outputfile=None, overwrite=False):
        """Load data from a numpy.ndarray and construct an astropy.io.fits HDU list

        To provide a uniform data type for all DKIST instruments the follow conventions are enforced:

        1. The PrimaryHDU can (and should) have relevant header information
        2. The PrimaryHDU will have no data
        3. ImageHDU's must have data with 2 dimensions

        A consequence of point 3 is that an extra, third dimension in the input array will be distributed across
        multiple ImageHDU's. Input arrays of dimensionality > 3 are not currently supported.

        If an output file is not specified then a temporary FITS file will be created to take advantage of
        astropy.io.fits' memmap ability. In this case the FitsData object will be completely lost when it is no
        longer needed. To save the object use the .write_out() method prior to deletion.

        Parameters
        ----------
        array : numpy.ndarray
            Array to be loaded into the FitsData object. Must have 2 or 3 dimensions
        outputfile : str
            If specified, a FITS file will be saved to this path and then reloaded. If None then a temporary file
            will be created.
        overwrite : bool
            If True the outputfile specified will be overwritten.
        """
        hdulist = [pyfits.PrimaryHDU()]
        hdulist[0].header['FRMARR'] = True
        if len(array.shape) == 3:
            for i in range(array.shape[2]):
                hdulist.append(pyfits.ImageHDU(array[:, :, i]))
        elif len(array.shape) > 3 or len(array.shape) < 2:
            raise TypeError('FitsData currently only supports initialization with ndarrays of shape 2 or 3')
        else:
            hdulist.append(pyfits.ImageHDU(array))

        # Now write to a file so we don't keep everything in memory
        if not outputfile:
            outputfile = '{}/tmp{}.fits'.format(TMPDIR, ''.join(str(time.time()).split('.')))
            self.is_tmp = True

        pyfits.HDUList(hdulist).writeto(outputfile, overwrite=overwrite)

        # And then simply load the resulting file
        self.load_from_file(outputfile)
    


    [docs]    def flush(self):
        """Flush the current data from memory and reload memmap from the disk
        """
        if self.is_tmp:
            outputfile = self.tmp_file
        else:
            outputfile = '{}/tmp{}.fits'.format(TMPDIR, ''.join(str(time.time()).split('.')))

        print('{}: Flushing {} to {}'.format(tag(), self, outputfile))

        self.write_out(outputfile, True)
        for d in self.data_hdus:
            del d.data
        self.load_from_file(outputfile)
        self.is_tmp = True
        self.tmp_file = outputfile
    


    [docs]    def write_out(self, filename, overwrite=True):
        """Save the current object to a FITS file

        Parameters
        ----------
        filename : str
            The location of the resulting FITS file

        overwrite :
            If True, overwrite the filename if it exists. Otherwise complain about a pre-existing file.
        """
        out_hdu_list = pyfits.HDUList()
        loaded_hdus = ['data' in h.__dict__ for h in self.hdulist]
        for i, h in enumerate(self.hdulist):
            h._do_not_scale_image_data = True
            if h.data is None:
                h._do_not_scale_image_data = False
                out_hdu_list.append(h)
                continue
            ih = pyfits.ImageHDU()
            out_hdu_list.append(ih)
            data = np.atleast_3d(h.data)
            if data.shape[0] != 1 and data.shape[2] == 1:
                data = np.moveaxis(data, 2, 0)
            out_hdu_list[-1].data = data
            out_hdu_list[-1].header = h.header
            if not loaded_hdus[i]:
                del h.data
            h._do_not_scale_image_data = False

        out_hdu_list.writeto(filename, overwrite=overwrite, output_verify='silentfix')
    

    def _update_data_hdus(self):
        """Fill data_hdu list with only those HDUs that have data.

        To avoid actually loading the data we look at the required NAXIS header value. It will be 0 (False) for HDUs
        with no data. There are other (maybe better?) ways to do this.
        """
        self.data_hdus = pyfits.HDUList([h for h in self.hdulist if h.header['NAXIS']])

    @property
    def hdulist(self):
        """Just return the internal property"""
        return self._hdulist

    @hdulist.setter
    def hdulist(self, value):
        """Simply update the internal value and then make sure data_hdus gets updated too"""
        self._hdulist = value
        self._update_data_hdus()

    @property
    def data(self):
        """Make sure the _DataList object is up to date and return it"""
        self._data_list.data_hdus = self.data_hdus
        return self._data_list

    @property
    def headers(self):
        """A very nice convenience property to quickly access a list of the header objects"""
        return [h.header for h in self.hdulist]

        """Return the index of just the ImageHDUs that have a specific header keyword value

        Parameters
        ----------
        keyword : str
            The header keyword to search for

        value : str, int, float
            The value of the keyword to match

        Returns
        -------
        numpy.ndarray
            A 1d list of the indices that where data_hdus[idx].header[KEYWORD] == value
        """

        for i, h in enumerate(self.data_hdus):
            try:
                if h.header[keyword] == value:
                    idxs.append(i)
            except KeyError:
                pass

        return np.array(idxs)
    

class _DataList():
    """This class exists so that the actual data arrays in FitsData can be accessed without necessarily loading all
    arrays into memory. It also uses np.squeeze to ignore the dummy 3rd dimension in a numpy view sense. Thus, the
    benefits of memmaping are preserved as long as possible."""

    def __init__(self, allow3D=False):
        """Thre's only one property here, a list that will eventually contain pyfits HDU objects"""
        self.data_hdus = pyfits.HDUList()
        self.allow3D = allow3D

    def __del__(self):
        """Keep any loaded arrays out of memory when we're gone"""
        for h in self.data_hdus:
            del h.data

    def __getitem__(self, item):
        """Check that the shape of the actual data is expected and then return the squeezed view of the requested data
        """
        data = self.data_hdus[item].data

        if self.allow3D:
            return data

        if len(data.shape) == 3 and data.shape[0] != 1:
            raise TypeError('Data shape ({}) is wrong'.format(data.shape))

        return np.squeeze(data)

    def __setitem__(self, key, value):
        """Just direct access to underlying array"""
        self.data_hdus[key].data = value

    def __len__(self):
        """Needed so that we can stop using len(FitsData.data_hdus)"""
        return len(self.data_hdus)

        """Load files from a list of file names

        Each file is checked to make sure it is from the correct MRN (the one that this class was initialized with),
        but no checks are made against program task or dataset ID.

        Parameters
        ----------
        file_list : lst
            A list of paths to files to load
        """
        hdulist = []

        for f in file_list:
            h = pyfits.open(f)

            # Let's just do a double-check that the slit position is correct
            if MosiacRepeatNumber(h[0].header) != self.MRN:
                h.close()
                del h
                raise ValueError('File {} is not from the correct MRN ({})'.format(f, self.MRN))

            hdulist.append(h[0])
            self.file_list.append(f)

        self.hdulist = pyfits.HDUList([pyfits.PrimaryHDU(data=None, header=hdulist[0].header.copy())] + hdulist)

    [docs]    def read_directory(self, data_dir, suffix='*.fits', dataset_ID=''):
        """Read all files in a directory and assign all files a MRN identifier. Then populate a dictionary that maps
        mosaic/repeat number to a list of files corresponding to that MRN.

        Parameters
        ----------
        data_dir : str
            Path to a directory that will be searched for CS steps

        suffix : str
            File mask. The full mask will be data_dir/*SUFFIX

        dataset_ID : str
            Only check data from this data set. A check is made against each file's ID_004 header value. Can be a blank
            string to just load all data sets.
        """
        file_list = glob('{}/*{}'.format(data_dir, suffix))
        self.MRN_file_dict = {}

        for f in file_list:
            h = pyfits.open(f)
            if h[0].header['DKIST004'] == 'Observation' and (dataset_ID is '' or h[0].header['ID___004'] == dataset_ID):
                MRN = MosiacRepeatNumber(h[0].header)
                try:
                    self.MRN_file_dict[MRN].append(f)
                except KeyError:
                    self.MRN_file_dict[MRN] = [f]

            h.close()
            del h

    [docs]class MosiacRepeatNumber():
    """This class allows for an easy way to quickly compare SPEC-0122 style headers to determine if they come from the
    same DL-NIRSP mosaic/repeat number (MRN). The smallest possible DL data unit is defined uniquely by a) its mosaic
    location in the FOV, and b) its repeat number. For example, if the Instrument Program requires a 2x2 mosaic of the
    FOV to be repeated 3 times then there will be 12 unique data units. Each data unit is defined by the vector:

    (DSPSRepeatNumber, CurrentSpatialStepX, CurrentSpatialStepY)

    This class is that vector.

    This class can be sorted and is hashable for use in dictionaries.
    """

    def __init__(self, header):
        """Initialize and read the MRN from a SPEC-0122 FITS header.

        Parameters
        ----------
        header : `astropy.io.fits.header.Header`
            Object containing a SPEC-0122 compliant header

        """
        self.current_repeat = header['DKIST009']
        self.current_step_x = header['DLN__031']
        self.current_step_y = header['DLN__035']

    def __repr__(self):
        return 'DL-NIRSP repeat {} of spatial step ({}, {})'.format(self.current_repeat,
                                                                    self.current_step_x,
                                                                    self.current_step_y)

    def __eq__(self, other):
        if not isinstance(other, MosiacRepeatNumber):
            return False

        for item in ['current_repeat', 'current_step_x', 'current_step_y']:
            if getattr(self, item) != getattr(other, item):
                return False

        return True

    def __lt__(self, other):
        return (self.current_repeat, self.current_step_x, self.current_step_y) < \
               (other.current_repeat, other.current_step_x, other.current_step_y)

    def __hash__(self):
        return hash((self.current_repeat, self.current_step_x, self.current_step_y))


    [docs]    def get_str(self):
        """Condense the MRN vector into a string that can be used for file names and such"""
        return '{:04}-{:04}-{:04}'.format(self.current_repeat, self.current_step_x, self.current_step_y)

    [docs]    def read_directory(self, data_dir, suffix='*.fits'):
        """Read all files in a directory and assign all files a CS step identifier. Then sort the unique CS steps and
        populate a dictionary that maps step number to a list of files corresponding to that step.

        Parameters
        ----------
        data_dir : str
            Path to a directory that will be searched for CS steps

        suffix : str
            File mask. The full mask will be data_dir/*SUFFIX
        """
        file_list = glob('{}/*{}'.format(data_dir, suffix))
        tmp_dict = {}

        for f in file_list:
            h = pyfits.open(f)
            if h[0].header['DKIST004'] != 'PolCal':
                continue
            CSs = CSStep(h[0].header)
            try:
                tmp_dict[CSs].append(f)
            except KeyError:
                tmp_dict[CSs] = [f]

            h.close()
            del h

        sorted_steps = sorted(tmp_dict.keys())
        for i in range(len(sorted_steps)):
            self.CS_step_file_dict[i] = tmp_dict[sorted_steps[i]]
            self.CS_step_obj_dict[i] = sorted_steps[i]

    [docs]class CSStep():
    """This class allows for an easy way to quickly compare SPEC-0122 style headers to determine if they come from the
    same Calibration Sequence (CS) step. Each step in a CS is defined by the configuration of the GOS, namely the
    status of the polarizer and retarder (in or out) and the angle of each. Because some CS schemes call for some
    GOS configurations to repeat a check is also made against the observation time for each object; the defaults for
    the ViSP Pipeline define a maximum difference in time where two exposures are considered different CS steps
    regardless of GOS configuration.

    This class can also be sorted. In this case, only the observation time is taken into account.

    Finally, this class is hashable for use in dictionaries. The hash is only based on the GOS configuration so that
    all objects from the same CS step result in the same hash. Python also checks that dictionary keys are equal so any
    conflicts with repeated CS steps are avoided.
    """

    def __init__(self, header):
        """Initialize and read the GOS configuration and time from a SPEC-0122 FITS header.

        Parameters
        ----------
        header : `astropy.io.fits.header.Header`
            Object containing a SPEC-0122 compliant header

        """
        self.pol_in = header['PAC__001']
        self.pol_theta = header['PAC__002']
        self.ret_in = header['PAC__003']
        self.ret_theta = header['PAC__004']
        self.obs_time = Time(header['DATE-OBS'])

    def __repr__(self):
        return 'CC with Pol = {}:{}, Ret = {}:{}. Taken at {}'.format(self.pol_in,
                                                                      self.pol_theta,
                                                                      self.ret_in,
                                                                      self.ret_theta,
                                                                      self.obs_time.fits)

    def __eq__(self, other):
        """Two steps are equal if they have the same GOS configuration and are taken within some package-defined time
        of each other."""
        if not isinstance(other, CSStep):
            return False

        for item in ['pol_in', 'pol_theta', 'ret_in', 'ret_theta']:
            if getattr(self, item) != getattr(other, item):
                return False

        tdelt = abs(self.obs_time - other.obs_time)
        if tdelt.sec > float(defaults['Main']['max_CS_step_time_sec']):
            return False

        return True

    def __lt__(self, other):
        """Only based on time"""
        return self.obs_time < other.obs_time

    def __hash__(self):
        """Only based on GSO configuration so that all objects from the same CS step hash the same"""
        return hash((self.pol_in, self.pol_theta, self.ret_in, self.ret_theta))

        """
        Load data from a demodulation-style numpy.ndarray and construct an astropy.io.fits HDU list

        Because all demodulation matrices should have shape (X, Y, L, 4, M), (X, Y, 4, M) or (4, M), a check is made to

        A temporary FITS file will be created to take advantage of astropy.io.fits' memmap ability. Thie means this

        prior to deletion.

        Parameters
        ----------
        array : numpy.ndarray
            Array to be loaded into the FitsData object. Must have either 5, 4, or 2 dimensions.

        beam : int
            The DL-NIRSP beam that the specified demodulation matrix belongs to
        """
        if beam not in [1, 2]:
            raise IndexError('DL-NIRSP only has beams 1 and 2!')

        self.hdulist[0].header['FRMARR'] = True

        if len(array.shape) != 5 and len(array.shape) != 2 and len(array.shape) != 4:
            raise ValueError('DemodulationData must have 2, 4, or 5 dimensions')
        if array.shape[-2] != 4:
            raise ValueError('The Stokes dimension of DemodulationData must have length 4')
        else:
            self.hdulist[beam] = pyfits.ImageHDU(array, header=header)

        self.flush()

class _TaskData(FitsData):
    """_TaskData is a class template that is used to define specific *Data classes that are designed to only hold data
    from a specific Instrument Program Task (DKIST004). It inherits all of the functionality of `FitsData` and defines
    an extra method that loads data from a directory, provided it matches the class's task. After loading, this class
    behaves exactly like `FitsData`.
    """

    def __init__(self, data_dir=None, dataset_ID='', memmap=None, **kwargs):
        """Initialize the empty object and maybe load some data."""

        FitsData.__init__(self, load_obj=None, memmap=memmap)

        self.ref_dir = ''
        self.dataset_ID = ''
        self.file_list = []
        if data_dir is not None:
            self.load_from_dir(data_dir, dataset_ID=dataset_ID, **kwargs)

    def __repr__(self):
        repstr = '{} loaded from {}'.format(self.__class__.__name__, self.ref_dir)
        if self.dataset_ID:
            repstr += ' with dataset ID = {}'.format(self.dataset_ID)
        return repstr

    def load_from_dir(self, data_dir, suffix='.fits', dataset_ID='', maxopen=1000):
        """Load DHS-style data from a directory.

        It is assumed that all relevant files in the directory have SPEC-0122 headers and a single HDU. In other words,
        each file is exactly one SysOutput. Only files whose InstrumentProgramTask (DKIST004) is equal to this class's
        task attribute will be loaded.

        Parameters
        ----------
        data_dir : str
            Path to the directory containing data to load

        suffix : str
            Optional file mask. The full mask is DATA_DIR/*SUFFIX. Defaults to ".fits"

        dataset_ID : str
            Only files whose ID_004 header value matches this parameter will be loaded. Can be a blank string to load
            all data sets.
        """
        file_list = sorted(glob('{}/*{}'.format(data_dir, suffix)))
        hdulist = []

        i = 0
        for f in file_list:
            h = pyfits.open(f)
            if h[0].header['DKIST004'] == self.task and (dataset_ID is '' or h[0].header['ID___004'] == dataset_ID):
                if i > maxopen:
                    self.hdulist = hdulist
                    self.flush()
                    i = 0

                hdulist.append(h[0])
                self.file_list.append(f)
                i += 1

        if len(hdulist) > 1:
            hdulist = [pyfits.PrimaryHDU(data=None, header=hdulist[0].header)] + hdulist
            self.hdulist = pyfits.HDUList(hdulist)
            self.ref_dir = data_dir
        else:
            raise FileNotFoundError('{}: Did not find any {} images with {}/*{}'.format(tag(), self.task, data_dir, suffix))
        self.dataset_ID = dataset_ID

# Here we define a separate _TaskData class for each type of InstrumentProgramTask (DKIST004) we'll need. Any additions
# can simply be added to this list.
for task in ['Dark','Gain','Alignment','Observation']:
    new_class_name = '{}Data'.format(task)
    docstring = """{} is an extension of `FitData` designed to only hold data from the {} InstrumentProgramTask
                (DKIST004). It inherits all of the functionality of `FitsData` and defines an extra method that loads
                data from a directory, provided it's DKIST004 value is '{}'.""".format(new_class_name, task, task)

    # This is some deep python-fu. type() is a metaclass that creates classes based on a BaseClass and dictionary of
    # attributes
    new_class = type(new_class_name, (_TaskData,), {'task': task, '__doc__': docstring})

    # Then we set the new class as one of this module's attributes
    setattr(sys.modules[__name__], new_class_name, new_class)

# And finally, clean up our mess
del new_class, new_class_name, docstring
